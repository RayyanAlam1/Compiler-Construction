# NumPat Compiler Test Case 8: Nested Function Calls
# Tests: Function calls within function arguments, call chain

int square(x) {
    return x * x
}

int addTwo(x) {
    return x + 2
}

int multiplyByThree(x) {
    return x * 3
}

# Nested calls: multiplyByThree(addTwo(square(3)))
# square(3) = 9, addTwo(9) = 11, multiplyByThree(11) = 33
print square(3)
print addTwo(9)
print multiplyByThree(11)
print multiplyByThree(addTwo(square(3)))
