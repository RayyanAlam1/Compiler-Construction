# NumPat Compiler Test Case 7: Function with Conditional Logic
# Tests: If-else inside function, parameter usage, multiple returns

int max(a, b) {
    if a > b {
        return a
    }
    else {
        return b
    }
}

int min(a, b) {
    if a < b {
        return a
    }
    else {
        return b
    }
}

print max(10, 5)
print max(3, 8)
print min(10, 5)
print min(3, 8)
